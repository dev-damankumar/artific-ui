export const Template=(args) => {
	const message=useMessage()
	message.config({
		dismissible: true
	})
	return <>
		<Button onClick={() => {
			message.success('Hi ! This is a message').hold()
		}}>
			Success
		</Button>
		<br/>
		<Button onClick={() => {
			message.info('Hi ! This is a message').hold()
		}}>
			Info
		</Button>
		<br/>
		<Button onClick={() => {
			message.error('Hi ! This is a message').hold()
		}}>
			Danger
		</Button>
		<br/>
		<Button onClick={() => {
			message.warning('Hi ! This is a message').hold()
		}}>
			Warning
		</Button>
	</>
};
